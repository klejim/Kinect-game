Créer une interface facilitant l'interprétation des données envoyées par la Kinect
squelette-defauts.cpp

Cet exemple montre la façon d'utiliser l'interface afin de récupérer les informations relatives à la qualité de la détection du squelette et les afficher dans une fenêtre SFML. Par souci de simplifier au maximum le code, le cas où plusieurs squelettes sont présents n'est pas pris en compte.

#include "SFML/Graphics.hpp"
#include <stdexcept>
#include "Kinect.hpp"
int main()
{
// on initialise l'interface
KtI::KinectInterface KInterface(KtI::INITIALIZE_SKELETON);
sf::RenderWindow window(sf::VideoMode(640, 480), "Defauts de squelette");
long angle(0);
// le squelette sera représenté par un tableau de cercles
sf::CircleShape articulations[KtI::SKELETON_JOINTS_COUNT];
for (int i = 0; i < 20; i++)
{
articulations[i].setFillColor(sf::Color::Red);
articulations[i].setRadius(8);
}
// on lance l'acquisition des trames de données de la Kinect
KInterface.start();
while (window.isOpen())
{
// on crée un évènement KinectEvent et entre dans la boucle d'évènement de l'interface
KtI::KinectEvent kEvent;
while (KInterface.pollEvent(kEvent))
{
if (kEvent.type == KtI::KinectEvent::SkeletonMoved)
{
// mise à jour de la position du squelette
for (int i = 0; i < 20; i++)
{
sf::Vector2f pos;
pos.x = kEvent.skeleton.positions[i].x;
pos.y = kEvent.skeleton.positions[i].y;
articulations[i].setPosition(pos);
}
}
// si la qualité de la détection n'est pas parfaite, on affichera en rouge les articulation dont les positions
// sont déduites
if (kEvent.type == KtI::KinectEvent::CannotSeeUser)
{
sf::Color tmp;
for (int i = 0; i < KtI::SKELETON_JOINTS_COUNT; i++)
{
switch (kEvent.quality.jointstates[i])
{
case KtI::POSITION_INFERRED:
tmp = sf::Color::Red; break;
case KtI::POSITION_TRACKED:
tmp = sf::Color::Blue; break;
case KtI::NOT_TRACKED:
tmp = sf::Color::White; break;
default:break;
}
articulations[i].setFillColor(tmp);
}
}
}
// fin de la boucle d'évènement de l'interface
sf::Event event;
while (window.pollEvent(event))
{
if (event.type == sf::Event::Closed)
window.close();
}
window.clear();
for (int i = 0; i < 20; i++)
{
window.draw(articulations[i]);
}
window.display();
}
// avant de terminer, on indique à l'interface qu'elle peut arrêter de recevoir les trames de la Kinect
KInterface.stop();
return 0;
}
