183 void KinectInterface::processFrames()
184 {
188  stopProcessing = false;
189  // les nouvelles trames qui seront traitées ce tour de boucle (les membres de la classe sont les trames déjà traitées).
190  NUI_SKELETON_FRAME newSkeletonFrame;
191  NUI_IMAGE_FRAME newDepthFrame;
192  NUI_INTERACTION_FRAME newInteractionFrame;
193  int dbg_cpt = 0;
194  LONG angle; //angle que forme le capteur par rapport à l'horizontale
196  m_kinect->NuiCameraElevationSetAngle(5);
198  while (!stopProcessing)
199  {
200  // modification de l'angle du capteur
201 
202  // Avant de traiter les nouvelles trames, on vérifie s'il est nécessaire de réinitialiser la reconnaissance de squelette
203  // Cette décision dépend des états du pointeur m_tracker et de la variable ternaire m_skeletonTrackingState.
204  // L'initialisation de base (celle du constructeur) ayant marchée, on ne vérifie pas que celle-ci fonctionne.
205  if (m_kinect->NuiInitializationFlags() & NUI_INITIALIZE_FLAG_USES_SKELETON)
206  {
207  // on réinitialise uniquement si la variable ternaire ne vaut pas MEM, pour ne pas avoir à le faire à toutes
208  // les itérations.
209  if (m_skeletonTrackingState == trinary::SET && m_tracker != nullptr) // tracker == nullptr est impossible dans ce cas
210  {
211  initializeSkeletonTracking(NUI_INITIALIZE_FLAG_USES_SKELETON);
212  m_skeletonTrackingState = trinary::MEM;
213  }
214  else if (m_skeletonTrackingState == trinary::RESET && m_tracker == nullptr) //tracker==nullptr est impossible
215  {
216  // l'état de m_skeletonTrackingState est directement pris en compte par initializeSkeletonTracking, donc on
217  // se contente de l'appeler à nouveau.
218  initializeSkeletonTracking(NUI_INITIALIZE_FLAG_USES_SKELETON);
219  m_skeletonTrackingState = trinary::MEM;
220  }
221  }
222 
223  // profondeur
224  if (WaitForSingleObject(m_nextDepthFrame, 0) == WAIT_OBJECT_0)
225  {
226  m_kinect->NuiImageStreamGetNextFrame(m_depthStream, INFINITE, &newDepthFrame);
227  processDepthFrame(newDepthFrame);
228  //
229  ResetEvent(m_nextDepthFrame);
230  m_depthFrame = newDepthFrame;
231  m_kinect->NuiImageStreamReleaseFrame(m_depthStream, &newDepthFrame);
232  }
233 
234  // skeleton
235  if (WaitForSingleObject(m_nextSkeletonFrame, 0) == WAIT_OBJECT_0)
236  {
237  m_kinect->NuiSkeletonGetNextFrame(0, &newSkeletonFrame);
238  m_kinect->NuiTransformSmooth(&newSkeletonFrame, 0);
239 
240  // analyse de la trame actuelle et comparaison avec l'ancienne.
241  procesSkeletonFrame(newSkeletonFrame);
242  // Si l'application a validé la sélection manuelle des squelettes à traquer, on exécute la méthode pointée par m_tracker.
243  if (m_tracker != nullptr)
244  {
245  setTrackedSkeletons(newSkeletonFrame);
246  }
247  ResetEvent(m_nextSkeletonFrame);
248  m_skeletonFrame = newSkeletonFrame;
249  }
250  // interaction
251  if (WaitForSingleObject(m_nextInteractionFrame,0) == WAIT_OBJECT_0)
252  {
253  m_interactionStream->GetNextFrame(INFINITE, &newInteractionFrame);
254  processInteractionFrame(newInteractionFrame);
255  m_interactionFrame = newInteractionFrame;
256  ResetEvent(m_nextInteractionFrame);
257  }
258 
259 
260  }
261  return;

337 void KinectInterface::procesSkeletonFrame(NUI_SKELETON_FRAME &newSkeletonFrame)
338 {
339  // interaction
340  Vector4 acc;
341  m_kinect->NuiAccelerometerGetCurrentReading(&acc);
342  HRESULT hr = m_interactionStream->ProcessSkeleton(NUI_SKELETON_COUNT, newSkeletonFrame.SkeletonData, &acc, newSkeletonFrame.liTimeStamp);
343  for (int i = 0; i < NUI_SKELETON_COUNT; i++)
344  {
345  NUI_SKELETON_DATA data = newSkeletonFrame.SkeletonData[i];
346  if (data.eTrackingState == NUI_SKELETON_POSITION_ONLY && moved(data))
347  {
348  // création d'un nouvel évènement
349  KinectEvent newEvent;
350  newEvent.type = KinectEvent::UserMoved;
351  newEvent.user.id = data.dwTrackingID;
352  // on convertit les coordonnées squelettes en coordonnées images
353  NuiTransformSkeletonToDepthImage(data.Position, &newEvent.user.position.x, &newEvent.user.position.y,
354  NUI_IMAGE_RESOLUTION_640x480);
355  // on ajoute le nouvel évènement à la pile ou au buffer (géré par la fonction)
356  addEventToStack(newEvent);
357  }
358  else if (data.eTrackingState == NUI_SKELETON_TRACKED)
359  {
361  // nouvel évènement SkeletonMoved
362  if (moved(data))
363  {
364  KinectEvent newEvent;
365  newEvent.type = KinectEvent::SkeletonMoved;
366  newEvent.skeleton.id = data.dwTrackingID;
367  for (int j = 0; j < NUI_SKELETON_POSITION_COUNT; j++)
368  {
369  // on convertit les coordonnées squelettes en coordonnées images
370  NuiTransformSkeletonToDepthImage(data.SkeletonPositions[j], &newEvent.skeleton.positions[j].x,
371  &newEvent.skeleton.positions[j].y, NUI_IMAGE_RESOLUTION_640x480);
372  }
373  // ajout à la pile d'évènements
374  addEventToStack(newEvent);
375  }
376 
377  // qualité de la détection
378  // on crée l'évènement et parcourt le détail pour le remplir
379  // si tout est parfait, on n'ajoute pas l'évènement à la pile
380  KinectEvent newEvent;
381  newEvent.type = KinectEvent::CannotSeeUser;
382  newEvent.quality.state = data.dwQualityFlags;
383  bool perfectQuality(true);
384  for (int n = 0; n < NUI_SKELETON_POSITION_COUNT; n++)
385  {
386  newEvent.quality.jointstates[n] = data.eSkeletonPositionTrackingState[n];
387  if (data.eSkeletonPositionTrackingState[n] != NUI_SKELETON_POSITION_TRACKED)
388  {
389  perfectQuality = false;
390  }
391  }
392  // ajout à la pile d'évènement
393  if (!perfectQuality)
394  {
395  addEventToStack(newEvent);
396  }
397 
398  }
399  }
400  return;

403 PositionType KinectInterface::getPosition(KinectEvent &data)
404 {
405  const float PI = 3.14152;
406  auto pos = data.skeleton.positions;
407  const int O = 0, BG(1), BD(2), JG(3), JD(4);
408  Vector4 vectors[5];
409  /* correspondance indice/axe
410  0 : axe central
411  1 : bras gauche
412  2 : bras droit
413  3 : jambe gauche
414  4 : jambe droite
415  */
416  for (int i = 0; i < 5; i++)
417  {
418  vectors[i].x = std::abs(pos[i+3*i].x - pos[i+3*i+2].x);
419  vectors[i].y = std::abs(pos[i+3*i].y - pos[i+3*i+2].y);
420  // norme
421  vectors[i].w = std::sqrt(std::pow(vectors[i].x, 2) + pow(vectors[i].x, 2));
422  }
423  // formule déduite de la logique suivante:
424  /*vectors[0].x = std::abs(pos[SKELETON_SHOULDER_CENTER].x - pos[SKELETON_HIP_CENTER].x);
425  vectors[0].y = std::abs(pos[SKELETON_SHOULDER_CENTER].y - pos[SKELETON_HIP_CENTER].y);
426 
427  vectors[1].x = std::abs(pos[SKELETON_WRIST_LEFT].x - pos[SKELETON_SHOULDER_LEFT].x);
428  vectors[1].y = std::abs(pos[SKELETON_WRIST_LEFT].y - pos[SKELETON_SHOULDER_LEFT].y);

431  vectors[2].y = std::abs(pos[SKELETON_WRIST_RIGHT].y - pos[SKELETON_SHOULDER_RIGHT].y);
432 
433  vectors[3].x = std::abs(pos[SKELETON_HIP_LEFT].x - pos[SKELETON_ANKLE_LEFT].x);
434  vectors[3].y = std::abs(pos[SKELETON_HIP_LEFT].y - pos[SKELETON_ANKLE_LEFT].y);
435 
436  vectors[4].x = std::abs(pos[SKELETON_HIP_RIGHT].x - pos[SKELETON_ANKLE_RIGHT].y);
437  vectors[4].y = std::abs(pos[SKELETON_HIP_RIGHT].y - pos[SKELETON_ANKLE_RIGHT].y);
438  */
439  float angles[5];
440  /* correspondance indice/angle (tous par rapport à l'axe central sauf l'axe central qui est par rapport au sol):
441  0 : bras gauche
442  1 : bras droit
443  2 : jambe gauche
444  3 : jambe droite
445  4 : axe central / sol
446  */
447  float pdtvec(0);
448  for (int i = 0; i < 4; i++)
449  {
450  // produit vectoriel

452  // u.v = |u|*|v|*cos(a) = u.x*v.x + u.y*v.y => a = arccos(pdtvec/(|u|*|v|))
453  angles[i] = std::acos(pdtvec / (vectors[0].w * vectors[i].w));
454  }
455  // calcul de l'angle entre l'axe central et le sol (1,0) à part:
456  pdtvec = vectors[0].x;
457  angles[4] = std::acos(pdtvec / (vectors[0].w));
458  // maintenant que les angles sont calculés, il faut encore reconnaître la position
459  /*position d'initialisation : bras horizontaux et jambes verticales,
460  */
461  if (angles[4] == PI/2 && angles[0])
462  { }
463  return PositionType::INITIALIZING;
464 }

505 bool KinectInterface::moved(NUI_SKELETON_DATA &newData)const
506 {
507  bool moved(false);
508  switch (newData.eTrackingState)
509  {
510  case NUI_SKELETON_POSITION_ONLY:
511  // retrouver l'utilisateur correspondant dans la trame précédente
512  for (int i = 0; i < NUI_SKELETON_COUNT; i++)
513  {
514  NUI_SKELETON_DATA data = m_skeletonFrame.SkeletonData[i];
515  // vérifier s'il a bougé (on valide les mouvements les plus petits)
516  if (data.dwTrackingID == newData.dwTrackingID)
517  {
518  if (newData.Position.x != data.Position.x || newData.Position.y != data.Position.y
519  || newData.Position.z != data.Position.z)
520  {
521  moved = true;
522  }
523  }
524  }
525  case NUI_SKELETON_TRACKED:
526  // retrouver le squelette correspondant
527  for (int i = 0; i < NUI_SKELETON_COUNT; i++)
528  {
529  NUI_SKELETON_DATA data = m_skeletonFrame.SkeletonData[i];
530  // on vérifie s'il y a eu mouvement pour chaque articulation
531  for (int j = 0; j < NUI_SKELETON_POSITION_COUNT; j++)
532  {
533  if (newData.SkeletonPositions[j].x != data.SkeletonPositions[j].x
534  || newData.SkeletonPositions[j].y != data.SkeletonPositions[j].y
535  || newData.SkeletonPositions[j].z != data.SkeletonPositions[j].z)
536  {
537  moved |= true;
538  }
539  }
540  }
541  }
542  return moved;

581 struct Bitmap_data_size encode_bitmap(unsigned char bits[], int size, int bitsperpixel, int width, int height, char name[])
582 {
583  BITMAPFILEHEADER header;
584  BITMAPINFOHEADER info;
585  int bytesperpixel = bitsperpixel / 8;
586  /*int extrabytes_nb = 4 - (bytesperpixel * width) % 4;
587  int paddedsize = (width*bytesperpixel + extrabytes_nb) * height;
588  */
589  header.bfType = 0x4D42;
590  header.bfSize = sizeof(BITMAPFILEHEADER) + sizeof(BITMAPINFOHEADER) + width*height*bytesperpixel;
591  header.bfReserved1 = 0;
592  header.bfReserved2 = 0;
593  header.bfOffBits = 54;
594 
595  info.biSize = sizeof(BITMAPINFOHEADER);
596  info.biWidth = width;
597  info.biHeight = -height;
598  info.biPlanes = 1;
599  info.biBitCount = bitsperpixel;
600  info.biCompression = 0;
601  info.biSizeImage = width*height*bytesperpixel;
602  info.biXPelsPerMeter = 2835;
603  info.biYPelsPerMeter = 2835;
604  info.biClrUsed = 0;
605  info.biClrImportant = 0;
606 
607  struct Bitmap_data_size result;
608  result.data = new char[header.bfSize + info.biSizeImage];
609  result.size = header.bfSize + info.biSizeImage;
610 
611  memcpy(result.data, (char*)&header, sizeof(BITMAPFILEHEADER));
612  memcpy(result.data + sizeof(BITMAPFILEHEADER), &info, info.biSize);
613  memcpy(result.data + header.bfSize, bits, size);
614 
615  std::ofstream output;
616  output.open(name, std::ofstream::out | std::ofstream::binary);
617  output.write((char*)&header, sizeof(BITMAPFILEHEADER));
618  output.write((char*)&info, sizeof(BITMAPINFOHEADER));
619 
620  output.write((char*)bits, info.biSizeImage);
621 
622  output.close();
623  return result;
